using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Day4
{
    internal class MathExercise
    {
        public int add(int a, int b)
        {
            int sum = a + b;
            return sum;
        }
        public int sub(int a, int b)
        {
            int sub = a - b;
            return sub;
        }
        public int mul(int a, int b)
        {
            int mul = a * b;
            return mul;
        }
        public int div(int a, int b)
        {
            int div = a / b;
            return div;
        }
    }
}
